# vue filters

new Vue({
  // ...
  filters: {
    nullCheck: value => value || '-',
    nl2br: str => !str ? '' : str.replace(/(?:\r\n|\r|\n)/g, '<br>'),
    entityDecode: str => {
      // with vanilla.js
      if (!str) {
        return ''
      } else {
        var ta = document.createElement('textarea')
        ta.innerHTML = str
        return ta.value
      }
      // with jQuery
      return !str ? '' : $('<textarea />').html(str).text()
      // with Underscore.js
      return !str ? '' : _.unescape(str)
    }
  },
  // ...
})
