# String, Template literals

// String
Vue.component('my-comp', {
  template: '<div class="box"><strong>{{ name }}</strong> component</div>',
  data() {
    return {
      name: 'haksoo'
    }
  },
  methods: { /* ... */ },
});

// Template literals - use backtick (` `)
Vue.component('my-comp', {
  template: `
    <div class="box">
      <strong>{{ name }}</strong> component
    </div>`,
  data() { /* ... */ },
  methods: { /* ... */ },
});

// use render

  data() { /* ...*/ },
  methods: { /* ...*/ },
  render(createElement) {
    return createElement(
      'div',
        {
          attrs: {
            'class': 'box'
          },
        },
        /* ... */
    );
  }
