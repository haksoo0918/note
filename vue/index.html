# Vue 에서 .vue 파일에서 CSS 변수 사용하기 & 계산된 스타일 적용하기

특정 값에 의해 스타일에 변화를 주고 싶을때

  • CSS 변수를 활용해 변수값을 수정하는 방법: 넓은 부분에 적용할때 좋을듯.
  • 적용할 스타일을 정의해서 적용하는 방법: 좁은 부분에 적용할때 좋을듯.
<template>
  <!-- CSS 변수를 활용한 방법 -->
  <div class="progress" :style="cssVariables">
    <!-- 적용할 스타일 정의 -->
    <div class="bar" :style="progressBarStyle">{{ current }} / {{ max }}
  </div>
</template>

<script>
export default {
  name: '...',
  data () {
    return {
      max: 100,
      current: 88
    }
  },
  computed: {
    // CSS 변수를 활용한 방법
    cssVariables () {
      return {
        '--bar-width': (current / max) + '%'
      }
    },
    // 적용할 스타일 정의
    progressBarStyle () {
      return {
        'width': (current / max) + '%'
      }
    }
  },
}
</script>

<style lang="sass" scoped>
// CSS 변수를 활용한 방법
.bar
  width: var(--bar-width)
</style>
