# Get selected text from option in select

// JavaScript
// ues options (or children) - selectedIndex
var s = document.getElementById('select-id');
var value = s.options[s.selectedIndex].value;
var text = s.children[s.selectedIndex].text;

// jQuery
// :selected
var text = $('#select-id option:selected').text();

# array 중복 제거 방법

const array = ['🐈', 1, 2, '🐈', '🐈', 3];

// 1: Set
[...new Set(array)];

// 2: Filter
array.filter((item, index) => array.indexOf(item) === index);

// 3: Reduce
array.reduce((unique, item) => unique.includes(item) ? unique : [...unique, item], []);

// Result
// ['🐈', 1, 2, 3];

# 한글만 입력 받기

var name = document.getElementById('name');
var onlyKorean = function() {
  var pattern = /[a-z0-9]|[ \[\]{}()<>?|`~!@#$%^&*-_+=,.;:\"'\\]/g;
  this.value = this.value.replace(pattern, '');
};
name.addEventListener('keyup', onlyKorean);

# Convert an Array into an Object

// arrays are technically objects.
const arr = ['hello', 'world'];
arr instanceof Object; // true

// Object.keys(), Object.entries()
Object.keys(arr); // ['0', '1']
Object.entries(arr); // [ ['0', 'hello'], ['1', 'world'] ]

// using `Object.assign()`
const obj = Object.assign({}, arr);
obj instanceof Object; // true
Array.isArray(obj); // false

obj; // { '0': 'hello', '1': 'world' }
