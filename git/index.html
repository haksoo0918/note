# 되돌리기 - reset, revert

# reset

git reset [option] [commit hash]
1

# option

  • hard: 돌아간 커밋의 다음 커밋을 전부 지움
  • soft: 돌아간 커밋의 다음 커밋이 스테이징 상태
  • mixed: 돌아간 커밋의 다음 커밋이 스테이징 전 상태, 옵션의 기본값
git reset --hard a2bbb3c
git reset HEAD~3

# hard 로 되돌린 커밋을 원격 레포지토리에 push
git push oright +master
1

# revert

git revert [commit hash]
1
  • 실제 커밋으로 돌아가지는 않고 생태만 되돌린다. 커밋을 이어 진행한다.
git revert 2664ce3
1

# 차이

실제 커밋으로 돌아가느냐, 그때의 내용(상태)만 가져오느냐 차이도 있지만, 원격 레포지토리에 push한 상태라면 revert 추천.('force' 라는 무시무시한 옵션이 있긴 함)

# 파일 추적 중지, gitignore 변경사항 적용하기

우선 파일을 추적하지 않으려면 gitignore 를 사용하면 된다.
문제는 이미 추적중인 파일을 gitignore 에 추가하거나 할때 일어난다.
gitignore 추적하지 않을 파일을 등록하고 관련된 파일을 일일이 제외시켜도 되지만,

git rm --cached [file]
1

심플하게 전체 추적을 리셋해도 된다.

git rm -r --cached .
git add .
git commit -m 'fixed untracked files'
1
2
3

하지만 gitignore 를 수정하지 않고, 추적에서 제외하고 싶은 경우도 있다.
대체로 설정파일을 개인 로컬에서 다르게 적용하고 싶은 경우가 되겠다.

git update-index --assume-unchanged [file]
1

위 방법을 사용하면 된다.

git update-index --no-assume-unchanged [file]
1

위 명령어로 추적을 재기할수 있다.
unchanged 로 지정된 파일은 아래 명령을 이용하면 된다.

git ls-files -v | grep ^h
1

# pull 을 위한 stash

pull 할때 merge 할때 modified(or add ?) 상태에 있으면 에러를 뛰운다. commit 하거나 stash 하란다.

# stash
$ git stash
$ git pull
$ git stash pop

# 한번에
$ git stash && git pull && git stash pop

# remote branch 가져오기

원격 레포지토리에 있는 브런치 확인 및 가져오기

# 원격 브런치 갱신
git remote update

# 브런치 확인
# 원격
git branch -r
# 로컬 및 원격 전부
git branch -a

# 원격 브런치 가져오기
git checkout -t [브런치 이름]

# 원격 브런치 pull 없이 소스 확인 - 수정(push) 불가
