# VuePress

VuePress 사용하면서 적용했던 내용들

# 설정

// file: .vuepress/config.js

module.exports = {
  // favicon
  head: [
    ['link', { rel: 'shortcut icon', type: 'image/x-icon', href: '/favicon.png' }]
  ],
  // 테마 설정
  themeConfig: {
    // sidebar
    // '/' 는 맨 마지막에 정의한다. - fallback
    // ref: https://vuepress.vuejs.org/theme/default-theme-config.html#sidebar
    sidebar: {
      '/favorite/': [
        ['', 'Intro'],
        {
          title: 'Favorite',
          collapsable: false,
          children: [
            // ...
          ],
        },
      ],
      '/': [
        {
          title: 'Notes',
          collapsable: false,
          children: [
            // ...
          ]
        },
        {
          title: 'Document',
          children: [
            // ...
          ]
        }
      ],
    },
    // ...
  },
  // 언어 설정
  locales: {
    '/': {
      lang: 'ko-KR'
    }
  },
  // markdown 설정
  markdown: {
    // 라인넘버
    lineNumbers: true
  },
  // 플러그인
  plugins: [
    // 상단으로 스크롤 되는 버튼
    // 플러그인 설치 필요
    // ref: https://vuepress.vuejs.org/plugin/official/plugin-back-to-top.html
    '@vuepress/back-to-top',
    // 마지막 수정일 출력
    // 기본 테마 사용시 플러그인 설치되어 있음
    '@vuepress/last-updated'
  ]
};
